REPLACE WITH:
```javascript
// Fetch full recommendation for a single event by ID
// Calls through MCP tool registry — no bare makeRequest
async function fetchEventRecommendation(eventId) {
    if (!apiToken || !eventId) return null;
    const tool = server.tools.get('get_event_recommendation');
    if (!tool) return null;
    const result = await tool.handler({ eventId: String(eventId) });
    return result.success ? result.event : null;
}
```

### Step 2: Add new tool to apiServiceTools array
Add AFTER the batch_post_queue tool entry (before the closing bracket of apiServiceTools):

```javascript
    {
        name: "get_event_recommendation",
        description: "Fetch full recommendation for a single irrigation event by event ID.",
        inputSchema: {
            type: "object",
            properties: {
                eventId: { type: "string", description: "CropManage irrigation event ID" }
            },
            required: ["eventId"]
        },
        handler: async (args) => {
            if (!apiToken) return { success: false, message: 'No token available' };
            if (!args.eventId) return { success: false, message: 'eventId required' };
            const options = {
                hostname: API_BASE_URL,
                path: `/${API_VERSION}/irrigation-events/${args.eventId}.json`,
                method: 'GET',
                headers: { 'Authorization': `Bearer ${apiToken}` }
            };
            try {
                const response = await makeRequest(options);
                if (response.statusCode === 200) {
                    return { success: true, event: response.data };
                }
                return { success: false, message: `Event fetch failed: ${response.statusCode}` };
            } catch (error) {
                return { success: false, message: error.message };
            }
        }
    }
```
